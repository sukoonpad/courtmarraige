  // Re-init FAQs
  initFAQs();
}

function getPageId(href) {
  if (!href) return null;
  // Clean href
  var h = href.replace(window.location.origin, '').replace(/^https?:\/\/[^\/]+/, '');
  // Check exact map
  if (PAGE_MAP[h]) return PAGE_MAP[h];
  // Check filename
  for (var key in PAGE_MAP) {
    if (h.indexOf(key) !== -1 && key !== 'index.html' && key !== './' && key !== '/' && key !== '#') {
      return PAGE_MAP[key];
    }
  }
  return null;
}

// Intercept all internal links
document.addEventListener('click', function(e) {
  var link = e.target.closest('a[href]');
  if (!link) return;
  var href = link.getAttribute('href');
  if (!href) return;
  
  // Skip external links, tel, mailto, whatsapp
  if (href.startsWith('tel:') || href.startsWith('mailto:') || href.startsWith('https://wa.me') || href.startsWith('http')) return;
  
  // Check hash-only links (like #services, #pricing etc) — scroll within current page
  if (href.startsWith('#') && href.length > 1) {
    // If hash points to a page map item
    var pageKey = href.substring(1) + '.html';
    if (PAGE_MAP[pageKey]) {
      e.preventDefault();
      showPage(PAGE_MAP[pageKey]);
      return;
    }
    // Otherwise let it scroll naturally within page
    return;
  }
  
  var pageId = getPageId(href);
  if (pageId) {
    e.preventDefault();
    showPage(pageId);
    // Handle anchor within page (e.g. /#pricing)
    if (href.indexOf('#') !== -1) {
      var anchor = href.split('#')[1];
      setTimeout(function() {
        var el = document.getElementById(anchor);
        if (el) el.scrollIntoView({behavior: 'smooth'});
      }, 150);
    }
  }
});

// Handle back/forward
window.addEventListener('popstate', function(e) {
  if (e.state && e.state.page) showPage(e.state.page);
});

/* ============================================================
   FAQ — works across all pages
   ============================================================ */
function initFAQs() {
  document.querySelectorAll('.faq-q').forEach(function(q) {
    // Remove old listeners by cloning
    var newQ = q.cloneNode(true);
    q.parentNode.replaceChild(newQ, q);
  });
  document.querySelectorAll('.faq-q').forEach(function(q) {
    q.addEventListener('click', function() {
      var item = q.closest('.faq-item');
      if (!item) return;
      var isOpen = item.classList.contains('open');
      // Close all in same container
      var container = item.parentElement;
      container.querySelectorAll('.faq-item').forEach(function(i) { i.classList.remove('open'); });
      if (!isOpen) item.classList.add('open');
    });
  });
}
initFAQs();

/* ============================================================
   SCROLL REVEAL — works across all pages
   ============================================================ */
function initReveals() {
  var obs = new IntersectionObserver(function(entries) {
    entries.forEach(function(e) {
      if (e.isIntersecting) { e.target.classList.add('visible'); obs.unobserve(e.target); }
    });
  }, {threshold: .07});
  document.querySelectorAll('.reveal:not(.visible)').forEach(function(el) { obs.observe(el); });
}
initReveals();
