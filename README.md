          ctx.moveTo(proj[i].x,proj[i].y);

          ctx.strokeStyle='rgba(212,175,55,'+a+')';
          ctx.lineWidth=0.5;
          ctx.stroke();
        }
      }
    }
  }

  function animate(){
    ctx.clearRect(0,0,W,H);
    // Deep space gradient
    var grd=ctx.createRadialGradient(W/2,H/2,0,W/2,H/2,Math.max(W,H)/1.5);
    grd.addColorStop(0,'rgba(10,31,61,0.15)');
    grd.addColorStop(1,'rgba(3,9,18,0)');
    ctx.fillStyle=grd;
    ctx.fillRect(0,0,W,H);
    drawConnections();
    for(var i=0;i<particles.length;i++){particles[i].update();particles[i].draw();}
    requestAnimationFrame(animate);
  }
  animate();

  // Mouse parallax
  var mx=W/2,my=H/2;
  document.addEventListener('mousemove',function(e){
    mx=e.clientX;my=e.clientY;
    particles.forEach(function(p){
      p.vx+=(mx-W/2)*0.000005;
      p.vy+=(my-H/2)*0.000005;
    });
  });
})();
