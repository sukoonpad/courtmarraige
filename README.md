<script>
// POPUP TIMER
let sec = 15;
const secEl = document.getElementById('sec-count');
const barEl = document.getElementById('popup-bar');

function closePopup(){
  const p = document.getElementById('popup-overlay');
  if(p){ p.style.opacity='0'; p.style.transition='opacity 0.3s'; setTimeout(()=>p.remove(),300); }
}

const timer = setInterval(()=>{
  sec--;
  if(secEl) secEl.textContent = sec;
  if(barEl) barEl.style.width = (sec/15*100)+'%';
  if(sec<=0){ clearInterval(timer); closePopup(); }
},1000);

// Close popup on overlay tap (outside box)
document.getElementById('popup-overlay').addEventListener('click', function(e){
  if(e.target === this) closePopup();
});

// FAQ ACCORDION — keyboard accessible
document.querySelectorAll('.faq-q').forEach(q=>{
  const toggle = () => {
    const item = q.parentElement;
    const isOpen = item.classList.contains('open');
    document.querySelectorAll('.faq-item').forEach(i=>{
      i.classList.remove('open');
      i.querySelector('.faq-q').setAttribute('aria-expanded','false');
    });
    if(!isOpen){
      item.classList.add('open');
      q.setAttribute('aria-expanded','true');
    }
  };
  q.addEventListener('click', toggle);
  q.addEventListener('keydown', e => { if(e.key==='Enter'||e.key===' '){ e.preventDefault(); toggle(); } });
});

// FORM SUBMIT → WHATSAPP

  const phone = document.getElementById('f_phone').value.trim();

  window.location.href = 'https://wa.me/918506989134?text='+encodeURIComponent(msg);
});

// Smooth scroll for anchor links
document.querySelectorAll('a[href^="#"]').forEach(a => {
  a.addEventListener('click', e => {
    const target = document.querySelector(a.getAttribute('href'));
    if(target){ e.preventDefault(); target.scrollIntoView({behavior:'smooth', block:'start'}); }
  });
});
